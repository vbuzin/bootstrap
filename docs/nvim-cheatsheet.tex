\sect{DIFF MODE\enspace{\scriptsize :windo\,diffthis to start}}
\begin{kt}
\kd{{]}c\,/\,{[}c}{next / prev diff hunk}
\kd{do\,/\,dp}{obtain diff from / put diff to other}
\kd{:diffupdate}{refresh diff highlighting}
\kd{:windo\,diffthis}{diff all visible windows}
\kd{:diffoff}{exit diff mode (current window)}
\kd{:diffoff!}{exit diff mode (all windows)}
\end{kt}
\snote{Open two files side-by-side: \kbd{:vs file2} then \kbd{:windo diffthis}. Or \kbd{nvim -d f1 f2} from CLI.}
