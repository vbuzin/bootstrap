\sect{HELIX MINDSET (start here)}
\begin{kt}
\kd{:tutor}{interactive built-in tutorial --- run it!}
\kd{SPC ?}{command palette / key discovery (use it constantly)}
\kd{selection first}{select target(s) before the verb (d/c/y etc)}
\kd{multi-sel core}{C/Alt-C add cursors; s select matches inside}
\kd{x}{select line (repeat x to extend downward)}
\kd{think ranges}{cursor = 1-char sel; act on any sel(s)}
\kd{no visual mode}{v = select/extend mode (motions now extend)}
\kd{avoid dw etc}{don't port Vim habits --- select then act}
\end{kt}
\snote{First thing: run \kbd{:tutor}. Use SPC ? constantly to discover.}

% ═══════════════════════════════════════
\sect{MODES}
\begin{kt}
\kd{Normal}{default --- navigation + commands}
\kd{Insert}{i / a / o / I / A / O --- minimal bindings}
\kd{Select (v)}{movements \textit{extend} selections instead of moving cursor}
\kd{goto (g)}{g prefix --- jumps, LSP goto, file ends}
\kd{match (m)}{m prefix --- brackets, surround, textobjects}
\kd{view (z / Z)}{z = temp view ops; Z = sticky view (Esc to exit)}
\kd{window (C-w)}{split / nav / swap windows (Vim-like)}
\kd{space (SPC)}{pickers + LSP + clipboard (leader-like)}
\kd{:}{command / ex mode (typable commands)}
\end{kt}
\snote{Esc always returns to Normal. Changes only checkpoint on Esc from Insert.}

\sect{SELECTION MANIPULATION}

\kd{Alt-x}{shrink to line bounds}
\kd{\%}{select entire file}

\kd{S}{split selection(s) on regex}

\kd{Alt--}{merge all selections}
\kd{Alt-\_}{merge consecutive selections}

\end{kt}
\snote{This is the heart of Helix. Master x, s, C, Alt-C, ; and \% early.}
